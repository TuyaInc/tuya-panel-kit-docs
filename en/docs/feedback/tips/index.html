    Tips

    Tips is a bubble box component.

    Code demo

    Bubble - top left

    import { Tips } from 'tuya-panel-kit'
    

    const [state, setState] = React.useState({
      topLeft: false,
      topCenter: false,
      topRight: false,
      bottomLeft: false,
      bottomCenter: false,
      bottomRight: false,
    });
    const bigTips = {
      width: 110,
      height: 64,
      borderRadius: 16,
    };
    

    <Tips
      show={state.topLeft}
      contentStyle={bigTips}
      bgColor="#333333"
      cornerPosition="topLeft"
    />
